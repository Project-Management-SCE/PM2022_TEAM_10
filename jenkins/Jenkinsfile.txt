
pipeline {
    agent {
        docker {
            image 'python:3.9'
        }
    }
    stages {
        stage('Build') {
            steps {
                withEnv(["HOME=${env.WORKSPACE}"]){
                    sh 'python -m pip install --upgrade pip'
                    sh 'pip install --user -r requirements.txt'
                    // sh 'pip install django mssql-django'
                    //sh 'python ProjectManagement/manage.py runserver'     RUN SERVER
                }
                sh 'python -m compileall ProjectManagement/'//manage.py'      COMPILE FILES
                //sh 'python -m py_compile manage.py'

            }
        }
        stage('Test') {
            steps {
                sh 'echo "Unit test + Integration + Covverage"'
                withEnv(["HOME=${env.WORKSPACE}"]){
                    // sh 'python ProjectManagement/manage.py test ./ProjectManagement'
                    sh 'python -m coverage run ProjectManagement/manage.py test ./ProjectManagement'
                    sh 'cd ProjectManagement; python -m coverage report'
                }
            }
        }
    }
}
